/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/

class Solution {
public:
    Node* copyRandomList(Node* head) {
    if(head == NULL) {
        return  head;
    }
     unordered_map<Node*, Node*> mp;
     Node* curr = head;
     Node* prev = NULL;
     Node* newHead = NULL;

     while(curr != NULL) {
        Node* newNode = new Node(curr->val);
        mp[curr] = newNode;
        if(newHead == NULL) {
            newHead = newNode;
            prev = newHead;
        }
        else {
            prev->next = newNode;
            prev = newNode;
        }
        curr = curr->next;
     }

     //filling random nodes now
     curr = head;
     Node* currNew = newHead;
     while(curr != NULL) {
        if(curr->random == NULL) {
            currNew->random = NULL;
        }
        else {
            currNew->random = mp[curr->random];
        }
        curr = curr->next;
        currNew = currNew->next;
     }

        return newHead;
    }
};
