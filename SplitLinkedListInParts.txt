/**
 * Derrinition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    vector<ListNode*> splitListToParts(ListNode* head, int k) {
        //calculating number of nodes in linked list
        int n = 0;
        ListNode* temp = head;
        while(temp != NULL) {
            n++;
            temp = temp->next;
        }  

        int fixNodeNumber = n / k;
        int extraNodes = n % k;

        vector<ListNode*> result(k, NULL);
        ListNode* prev = NULL;
        ListNode* curr = head;
        for(int i = 0; i < k; i++) {
            result[i] = curr;
            for(int j = 1; j <= fixNodeNumber + (extraNodes > 0 ? 1 : 0); j++) {
                prev = curr;
                curr = curr->next;
            }
            if(prev != NULL) {
                prev->next = NULL;
            }
            extraNodes--;
        }

        return result;
    }
};
